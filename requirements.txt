dearpygui               # GPU-accelerated immediate-mode widget panel (REQ-010)

# ---------------------------------------------------------------------------
# Testing
# ---------------------------------------------------------------------------
pytest                  # Unit and integration test framework (spec §6)
Pillow                  # Screenshot comparison for visual regression tests (spec §6)
